# 1. Update system & install Docker
sudo apt update && sudo apt upgrade -y
sudo apt install -y curl git

# Install Docker
curl -fsSL https://get.docker.com | sh
sudo systemctl enable docker
sudo systemctl start docker

# Install Docker Compose (v2 plugin)
sudo apt install -y docker-compose-plugin

# Install Node.js (for frontend build)
curl -fsSL https://deb.nodesource.com/setup_20.x | sudo -E bash -
sudo apt install -y nodejs

# 2. Open firewall ports
sudo ufw allow 22/tcp
sudo ufw allow 80/tcp
sudo ufw allow 443/tcp
sudo ufw allow 1883/tcp
sudo ufw allow 9001/tcp
sudo ufw --force enable

# 3. Clone the project
cd /projects
git clone https://github.com/TheVasuA/iotaps.git
cd iotaps

# 4. Copy your .env file (contains secrets — do NOT commit)
nano .env
# Paste the full .env content from your local d:\iotaps\.env

# 5. Build frontend
cd web
npm install
npm run build
cd ..

# 6. Build and start all services
docker compose build fastapi-api
docker compose up -d

# 7. Wait for DB to be healthy, then run migrations
sleep 15
docker exec -w /srv/app iotaps-api alembic upgrade head

# 8. Set Postgres password (if fresh volume)
docker exec iotaps-postgres psql -U iotaps -d iotaps -c "ALTER USER iotaps PASSWORD 'zzo0p5d3-7hmiGuxe3xWiyry1gZWu_J0';"

# 9. Verify everything is running
docker compose ps
curl http://localhost:8000/api/v1/health
